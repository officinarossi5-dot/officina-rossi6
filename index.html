function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback; let result; try { if (action === 'getOrdini') result = getOrdini(); else if (action === 'getProduzione') result = getProduzione(); else if (action === 'getDashboard') result = getDashboard(); else if (action === 'addOrdine') result = addOrdine(e.parameter); else if (action === 'addProduzione') result = addProduzione(e.parameter); else if (action === 'getOperatori') result = getOperatori(); else result = { error: 'azione non trovata' }; } catch(err) { result = { error: err.toString() }; } const json = JSON.stringify(result); if (callback) { return ContentService.createTextOutput(callback + '(' + json + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === 'Ordini') sheet.appendRow(['ID','Cliente','Rif','DataOrdine','DataConsegna','QtyTotale','Stato','Note','Fasi']); if (name === 'FasiOrdine') sheet.appendRow(['OrdineID','Fase','QtyRichiesta','QtyProdotta','Stato']); if (name === 'Produzione') sheet.appendRow(['ID','Data','Operatore','OrdineID','OrdineRif','Fase','Pezzi','OreImpegate','Note']); } return sheet; } function getOrdini() { const ordSheet = getSheet('Ordini'); const fasiSheet = getSheet('FasiOrdine'); const ordData = ordSheet.getDataRange().getValues(); const fasiData = fasiSheet.getDataRange().getValues(); if (ordData.length <= 1) return []; const oh = ordData[0]; const fh = fasiData.length > 1 ? fasiData[0] : []; const fasiRows = fasiData.length > 1 ? fasiData.slice(1) : []; return ordData.slice(1).map(r => { const o = {}; oh.forEach((k,i) => o[k] = r[i]); o.fasi = fasiRows .filter(fr => fr[0] === o.ID) .map(fr => { const f={}; fh.forEach((k,i) => f[k]=fr[i]); return f; }); return o; }); } function getProduzione() { const data = getSheet('Produzione').getDataRange().getValues(); if (data.length <= 1) return []; const h = data[0]; return data.slice(1).map(r => { const o={}; h.forEach((k,i)=>o[k]=r[i]); return o; }); } function getOperatori() { const prod = getSheet('Produzione').getDataRange().getValues(); if (prod.length <= 1) return []; const h = prod[0]; const rows = prod.slice(1).map(r => { const o={}; h.forEach((k,i)=>o[k]=r[i]); return o; }); const stats = {}; rows.forEach(r => { const op = r.Operatore; if (!op) return; if (!stats[op]) stats[op] = { nome:op, pezziTotali:0, oreTotali:0, sessioni:0 }; stats[op].pezziTotali += parseFloat(r.Pezzi) || 0; stats[op].oreTotali += parseFloat(r.OreImpegate) || 0; stats[op].sessioni++; }); return Object.values(stats).map(s => ({ ...s, pezziPerOra: s.oreTotali > 0 ? (s.pezziTotali / s.oreTotali).toFixed(2) : '—' })); } function getDashboard() { const ordini = getSheet('Ordini').getDataRange().getValues(); const prod = getSheet('Produzione').getDataRange().getValues(); const oggi = new Date().toDateString(); const r = { totOrdini:0, ordiniCompleti:0, ordiniInRitardo:0, pezziOggi:0, oreOggi:0 }; if (ordini.length > 1) { r.totOrdini = ordini.length - 1; ordini.slice(1).forEach(row => { const cons = row[4] ? new Date(row[4]) : null; if (row[6] === 'Completato') r.ordiniCompleti++; if (cons && cons < new Date() && row[6] !== 'Completato') r.ordiniInRitardo++; }); } if (prod.length > 1) { prod.slice(1).forEach(row => { if (row[1] && new Date(row[1]).toDateString() === oggi) { r.pezziOggi += parseFloat(row[6]) || 0; r.oreOggi += parseFloat(row[7]) || 0; } }); } return r; } function addOrdine(p) { const id = 'ORD-' + Date.now(); const fasi = JSON.parse(p.fasi || '[]'); getSheet('Ordini').appendRow([ id, p.cliente||'', p.rif||'', p.dataOrdine||'', p.dataConsegna||'', parseFloat(p.qtyTotale)||0, 'In corso', p.note||'', p.fasi||'[]' ]); const fasiSheet = getSheet('FasiOrdine'); fasi.forEach(f => { fasiSheet.appendRow([id, f.nome, parseFloat(f.qty)||0, 0, 'In attesa']); }); return { success: true, id }; } function addProduzione(p) { const ordineId = p.ordineId; const fase = p.fase; const pezzi = parseFloat(p.pezzi) || 0; const ore = parseFloat(p.ore) || 0; // Controlla quantità fase const fasiSheet = getSheet('FasiOrdine'); const fasiData = fasiSheet.getDataRange().getValues(); let faseRow = -1; for (let i = 1; i < fasiData.length; i++) { if (fasiData[i][0] === ordineId && fasiData[i][1] === fase) { faseRow = i; break; } } if (faseRow === -1) return { success:false, error:'Fase non trovata nell\'ordine' }; const qtyRichiesta = parseFloat(fasiData[faseRow][2]) || 0; const qtyGiaFatta = parseFloat(fasiData[faseRow][3]) || 0; const qtyDisponibile = qtyRichiesta - qtyGiaFatta; if (pezzi > qtyDisponibile) { return { success: false, error: '⚠ Quantità superata! Puoi inserire max ' + qtyDisponibile + ' pezzi per questa fase.' }; } // Aggiorna fase const nuovaQty = qtyGiaFatta + pezzi; fasiSheet.getRange(faseRow+1, 4).setValue(nuovaQty); fasiSheet.getRange(faseRow+1, 5).setValue(nuovaQty >= qtyRichiesta ? 'Completata' : 'In corso'); // Salva produzione const id = 'PROD-' + Date.now(); getSheet('Produzione').appendRow([ id, p.data||'', p.operatore||'', ordineId, p.rif||'', fase, pezzi, ore, p.note||'' ]); // Controlla se tutte le fasi completate → ordine completato const fasiOrdine = fasiData.slice(1).filter(r => r[0] === ordineId); const tutteComplete = fasiOrdine.every(r => { if (r[1] === fase) return nuovaQty >= qtyRichiesta; return r[4] === 'Completata'; }); if (tutteComplete) { const ordSheet = getSheet('Ordini'); const ordData = ordSheet.getDataRange().getValues(); for (let i = 1; i < ordData.length; i++) { if (ordData[i][0] === ordineId) { ordSheet.getRange(i+1, 7).setValue('Completato'); break; } } } return { success:true, id, qtyRimasta: qtyDisponibile - pezzi, tutteComplete }; }
